#############
#### SINGLE
#############
## 확인
tibero$ ps -ef |grep -E "tbsvr|tibero"

## 중지
tibero$ tbdown immediate

## 기동
tibero$ tbboot 

## 기동 중 오류 발생할 경우
-> 오류 발생할 경우
tibero$ tbdown clean
tibero$ tbboot

#############
#### TSC
#############
## 확인
primary tibero$ cmrctl show
primary tibero$ $ ps -ef |grep -E "tbsvr|tibero|tbcm"
standby tibero$ cmrctl show
standby observer$ cmrctl show
standby tibero$ $ ps -ef |grep -E "tbsvr|tibero|tbcm"

## 중지
standby root# su - observer
standby observer$ tbcmobs -d

primary root# su - tibero
primary tibero$ tbdown immediate
primary tibero$ tbcm -d

standby root# su - tibero
standby tibero$ tbdown immediate
standby tibero$ tbcm -d

## 기동
standby root# su - observer
standby tibero$ tbcmobs -b

primary root# su - tibero
primary tibero$ tbcm -b
primary tibero$ tbboot

standby root# su - tibero
standby tibero$ tbcm -b
standby tibero$ tbboot
